\usepackage{float}

\makeatletter
\let\oldlt\longtable
\let\endoldlt\endlongtable
\def\longtable{\@ifnextchar[\longtable@i \longtable@ii}
\def\longtable@i[#1]{
    \begin{table}[H]
    \onecolumn
    \vspace{-15pt}
    \begin{minipage}{0.47\textwidth}
    \oldlt[#1]
}
\def\longtable@ii{
    \begin{table}[H]
    \onecolumn
    \vspace{-15pt}
    \begin{minipage}{0.47\textwidth}
    \oldlt
}
\def\endlongtable{
    \bottomrule
    \endoldlt
    \end{minipage}
    \vspace{-10pt}
    \twocolumn
    \end{table}
}
\makeatother
